## ✅ Step 2: Create `reset.ts`

📁 `src/db/reset.ts`

```ts id="c7m4pa"
import { db } from '../db'; // adjust path if needed

async function resetDatabase() {
  console.log('🧹 Dropping all tables...');

  await db.execute(`DROP TABLE IF EXISTS ecom_nestjs_orderitems CASCADE;`);
  await db.execute(`DROP TABLE IF EXISTS ecom_nestjs_orders CASCADE;`);
  await db.execute(`DROP TABLE IF EXISTS ecom_nestjs_products CASCADE;`);
  await db.execute(`DROP TABLE IF EXISTS ecom_nestjs_users CASCADE;`);

  console.log('🧹 Dropping ENUM types...');

  await db.execute(`DROP TYPE IF EXISTS order_status CASCADE;`);
  await db.execute(`DROP TYPE IF EXISTS payment CASCADE;`);

  console.log('🧹 Dropping migration tables...');

  await db.execute(`DROP TABLE IF EXISTS drizzle_migrations;`);
  await db.execute(`DROP TABLE IF EXISTS drizzle._drizzle_migrations;`);

  console.log('✅ Database reset completed');
}

resetDatabase()
  .then(() => process.exit(0))
  .catch((err) => {
    console.error('❌ Reset failed:', err);
    process.exit(1);
  });
```
