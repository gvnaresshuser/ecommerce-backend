# ✅ 1. USERS TABLE

```sql
CREATE TABLE IF NOT EXISTS ecom_nestjs_users (
    user_id SERIAL PRIMARY KEY,
    email VARCHAR(100) NOT NULL UNIQUE,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(200),
    fullname VARCHAR(100) NOT NULL,
    google_id VARCHAR(100) UNIQUE,
    roles VARCHAR(10)[] NOT NULL DEFAULT ARRAY['customer'],
    address VARCHAR(200),
    city VARCHAR(100),
    state VARCHAR(100),
    country VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Case-insensitive unique email
CREATE UNIQUE INDEX IF NOT EXISTS idx_ecom_users_lower_email
ON ecom_nestjs_users (LOWER(email));

-- Case-insensitive unique username
CREATE UNIQUE INDEX IF NOT EXISTS idx_ecom_users_lower_username
ON ecom_nestjs_users (LOWER(username));
```

# ✅ 2. PRODUCTS TABLE

```sql
CREATE TABLE IF NOT EXISTS ecom_nestjs_products (
    product_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    slug VARCHAR(100) NOT NULL UNIQUE,
    price REAL NOT NULL,
    description TEXT NOT NULL,
    image_url VARCHAR,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

# ✅ 4. ORDERS TABLE

```sql
CREATE TABLE IF NOT EXISTS ecom_nestjs_orders (
    order_id SERIAL PRIMARY KEY,
    user_id INTEGER NOT NULL,
    status VARCHAR(20) NOT NULL,
    date TIMESTAMP DEFAULT CURRENT_DATE,
    amount REAL,
    total REAL,
    ref VARCHAR(100),
    payment_method payment,
    CONSTRAINT fk_ecom_user 
        FOREIGN KEY (user_id) 
        REFERENCES ecom_nestjs_users(user_id) 
        ON DELETE CASCADE
);
```

---

# ✅ 5. ORDER ITEMS TABLE

```sql
CREATE TABLE IF NOT EXISTS ecom_nestjs_orderitems (
    orderitem_id SERIAL PRIMARY KEY,
    order_id INT NOT NULL,
    product_id INT NOT NULL,
    quantity INT NOT NULL CHECK (quantity > 0),
    price NUMERIC(10, 2) NOT NULL CHECK (price >= 0),

    CONSTRAINT fk_ecom_order 
        FOREIGN KEY (order_id) 
        REFERENCES ecom_nestjs_orders(order_id) 
        ON DELETE CASCADE,

    CONSTRAINT fk_ecom_product 
        FOREIGN KEY (product_id) 
        REFERENCES ecom_nestjs_products(product_id) 
        ON DELETE CASCADE
);
```

# ✅ 3. PAYMENT ENUM (REUSABLE)

```sql
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'payment') THEN
        CREATE TYPE payment AS ENUM ('CASH', 'CARD', 'UPI');
    END IF;
END
$$;

DO $$
BEGIN
    IF NOT EXISTS (
        SELECT 1 FROM pg_enum 
        WHERE enumlabel = 'RAZORPAY' 
        AND enumtypid = 'payment'::regtype
    ) THEN
        ALTER TYPE payment ADD VALUE 'RAZORPAY';
    END IF;
END
$$;
```
