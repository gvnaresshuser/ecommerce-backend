Step 1: Create config file

📁 drizzle.config.ts

import type { Config } from 'drizzle-kit';

export default {
  schema: './src/db/schema/index.ts',
  out: './drizzle',
  driver: 'pg',
  dbCredentials: {
    connectionString: process.env.DATABASE_URL!,
  },
} satisfies Config;
Step 2: Add scripts

📁 package.json

{
  "scripts": {
    "db:generate": "drizzle-kit generate",
    "db:migrate": "drizzle-kit migrate"
  }
}

pnpm db:generate

Step 4: Run migration

============================= IMPORTANT NOTE ====================================
👉 NestJS modules are isolated

So:

OrdersService exists in OrdersModule

👉 But PaymentModule cannot see it unless:

✔ OrdersModule exports it
✔ PaymentModule imports OrdersModule

🚀 ✅ FIX (STEP-BY-STEP)
✅ STEP 1: Export OrdersService

📁 orders.module.ts

import { Module } from '@nestjs/common';
import { OrdersService } from './orders.service';
import { OrdersController } from './orders.controller';
import { CartModule } from '../cart/cart.module';

@Module({
  imports: [CartModule],
  controllers: [OrdersController],
  providers: [OrdersService],
  exports: [OrdersService], // ✅ VERY IMPORTANT
})
export class OrdersModule {}
✅ STEP 2: Import OrdersModule in PaymentModule

📁 payment.module.ts

import { Module } from '@nestjs/common';
import { PaymentController } from './payment.controller';
import { PaymentService } from './payment.service';
import { OrdersModule } from '../orders/orders.module'; // ✅
import { CartModule } from '../cart/cart.module'; // ✅

@Module({
  imports: [OrdersModule, CartModule], // ✅ ADD BOTH
  controllers: [PaymentController],
  providers: [PaymentService],
})
export class PaymentModule {}
🧠 Dependency Flow (IMPORTANT)
OrdersModule → exports OrdersService
PaymentModule → imports OrdersModule
PaymentController → can use OrdersService ✅

👉 Always remember:

Service in Module A
→ must be exported
→ Module B must import Module A
→ then usable in Module B
